1. –†–µ–∞–ª–∏–∑—É–π—Ç–µ —Å–ª—É—á–∞–π–Ω—É—é –ø—Ä–æ–≥—É–ª–∫—É —Å –ø–æ–º–æ—â—å—é –∫–æ–¥–∞ –Ω–∏–∂–µ:

    ```python
    def random_policy(m):
        return random.choice(list(actions))
    
    def walk(m,policy,start_position=None):
        n = 0 # number of steps
        # set initial position
        if start_position:
            m.human = start_position 
        else:
            m.random_start()
        while True:
            if m.at() == Board.Cell.apple:
                return n # success!
            if m.at() in [Board.Cell.wolf, Board.Cell.water]:
                return -1 # eaten by wolf or drowned
            while True:
                a = actions[policy(m)]
                new_pos = m.move_pos(m.human,a)
                if m.is_valid(new_pos) and m.at(new_pos)!=Board.Cell.water:
                    m.move(a) # do the actual move
                    break
            n+=1
    
    walk(m,random_policy)
    ```

## –§—É–Ω–∫—Ü–∏—è –Ω–∞–≥—Ä–∞–¥—ã

–ß—Ç–æ–±—ã —Å–¥–µ–ª–∞—Ç—å –Ω–∞—à—É –ø–æ–ª–∏—Ç–∏–∫—É –±–æ–ª–µ–µ —Ä–∞–∑—É–º–Ω–æ–π, –Ω–∞–º –Ω—É–∂–Ω–æ –ø–æ–Ω—è—Ç—å, –∫–∞–∫–∏–µ —à–∞–≥–∏ "–ª—É—á—à–µ" –¥—Ä—É–≥–∏—Ö. –î–ª—è —ç—Ç–æ–≥–æ –Ω–∞–º –Ω—É–∂–Ω–æ –æ–ø—Ä–µ–¥–µ–ª–∏—Ç—å –Ω–∞—à—É —Ü–µ–ª—å.

```python
move_reward = -0.1

end_reward = -10

def reward(m,pos=None):
    pos = pos or m.human
    if not m.is_valid(pos):
        return end_reward
    x = m.at(pos)
    if x==Board.Cell.water or x == Board.Cell.wolf:
        return end_reward
    if x==Board.Cell.apple:
        return goal_reward
    return move_reward
```

## Q-Learning

–ê–ª–≥–æ—Ä–∏—Ç–º, –∫–æ—Ç–æ—Ä—ã–π –º—ã —Ä–∞—Å—Å–º–æ—Ç—Ä–∏–º –∑–¥–µ—Å—å, –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è **Q-Learning**. –í —ç—Ç–æ–º –∞–ª–≥–æ—Ä–∏—Ç–º–µ –ø–æ–ª–∏—Ç–∏–∫–∞ –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç—Å—è —Ñ—É–Ω–∫—Ü–∏–µ–π (–∏–ª–∏ —Å—Ç—Ä—É–∫—Ç—É—Ä–æ–π –¥–∞–Ω–Ω—ã—Ö), –Ω–∞–∑—ã–≤–∞–µ–º–æ–π **Q-—Ç–∞–±–ª–∏—Ü–µ–π**. –û–Ω–∞ —Ñ–∏–∫—Å–∏—Ä—É–µ—Ç "–∫–∞—á–µ—Å—Ç–≤–æ" –∫–∞–∂–¥–æ–≥–æ –¥–µ–π—Å—Ç–≤–∏—è –≤ –∑–∞–¥–∞–Ω–Ω–æ–º —Å–æ—Å—Ç–æ—è–Ω–∏–∏.

–û–Ω–∞ –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è Q-—Ç–∞–±–ª–∏—Ü–µ–π, –ø–æ—Ç–æ–º—É —á—Ç–æ –µ—ë —É–¥–æ–±–Ω–æ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è—Ç—å –≤ –≤–∏–¥–µ —Ç–∞–±–ª–∏—Ü—ã –∏–ª–∏ –º–Ω–æ–≥–æ–º–µ—Ä–Ω–æ–≥–æ –º–∞—Å—Å–∏–≤–∞. –ü–æ—Å–∫–æ–ª—å–∫—É –Ω–∞—à–∞ –¥–æ—Å–∫–∞ –∏–º–µ–µ—Ç —Ä–∞–∑–º–µ—Ä—ã `width` x `height`, –º—ã –º–æ–∂–µ–º –ø—Ä–µ–¥—Å—Ç–∞–≤–∏—Ç—å Q-—Ç–∞–±–ª–∏—Ü—É —Å –ø–æ–º–æ—â—å—é –º–∞—Å—Å–∏–≤–∞ numpy —Ñ–æ—Ä–º—ã `width` x `height` x `len(actions)`: (–±–ª–æ–∫ –∫–æ–¥–∞ 6)

```python
Q = np.ones((width,height,len(actions)),dtype=np.float)*1.0/len(actions)
```
