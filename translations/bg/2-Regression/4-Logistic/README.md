### –ü—Ä–µ–¥–≤–∞—Ä–∏—Ç–µ–ª–Ω–∞ –æ–±—Ä–∞–±–æ—Ç–∫–∞ –Ω–∞ –¥–∞–Ω–Ω–∏: –∫–æ–¥–∏—Ä–∞–Ω–µ –Ω–∞ —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏—Å—Ç–∏–∫–∏ –∏ –µ—Ç–∏–∫–µ—Ç–∏

–ù–∞—à–∏—è—Ç –Ω–∞–±–æ—Ä –æ—Ç –¥–∞–Ω–Ω–∏ –∑–∞ —Ç–∏–∫–≤–∏ —Å—ä–¥—ä—Ä–∂–∞ —Ç–µ–∫—Å—Ç–æ–≤–∏ —Å—Ç–æ–π–Ω–æ—Å—Ç–∏ –∑–∞ –≤—Å–∏—á–∫–∏ —Å–≤–æ–∏ –∫–æ–ª–æ–Ω–∏. –†–∞–±–æ—Ç–∞—Ç–∞ —Å –∫–∞—Ç–µ–≥–æ—Ä–∏—á–Ω–∏ –¥–∞–Ω–Ω–∏ –µ –∏–Ω—Ç—É–∏—Ç–∏–≤–Ω–∞ –∑–∞ —Ö–æ—Ä–∞—Ç–∞, –Ω–æ –Ω–µ –∏ –∑–∞ –º–∞—à–∏–Ω–∏—Ç–µ. –ê–ª–≥–æ—Ä–∏—Ç–º–∏—Ç–µ –∑–∞ –º–∞—à–∏–Ω–Ω–æ –æ–±—É—á–µ–Ω–∏–µ —Ä–∞–±–æ—Ç—è—Ç –¥–æ–±—Ä–µ —Å —á–∏—Å–ª–∞. –ó–∞—Ç–æ–≤–∞ –∫–æ–¥–∏—Ä–∞–Ω–µ—Ç–æ –µ –º–Ω–æ–≥–æ –≤–∞–∂–Ω–∞ —Å—Ç—ä–ø–∫–∞ –≤ –µ—Ç–∞–ø–∞ –Ω–∞ –ø—Ä–µ–¥–≤–∞—Ä–∏—Ç–µ–ª–Ω–∞ –æ–±—Ä–∞–±–æ—Ç–∫–∞ –Ω–∞ –¥–∞–Ω–Ω–∏, —Ç—ä–π –∫–∞—Ç–æ –Ω–∏ –ø–æ–∑–≤–æ–ª—è–≤–∞ –¥–∞ –ø—Ä–µ–≤—ä—Ä–Ω–µ–º –∫–∞—Ç–µ–≥–æ—Ä–∏—á–Ω–∏—Ç–µ –¥–∞–Ω–Ω–∏ –≤ —á–∏—Å–ª–æ–≤–∏, –±–µ–∑ –¥–∞ –≥—É–±–∏–º –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏—è. –î–æ–±—Ä–æ –∫–æ–¥–∏—Ä–∞–Ω–µ –≤–æ–¥–∏ –¥–æ –∏–∑–≥—Ä–∞–∂–¥–∞–Ω–µ –Ω–∞ –¥–æ–±—ä—Ä –º–æ–¥–µ–ª.

–ó–∞ –∫–æ–¥–∏—Ä–∞–Ω–µ –Ω–∞ —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏—Å—Ç–∏–∫–∏ –∏–º–∞ –¥–≤–∞ –æ—Å–Ω–æ–≤–Ω–∏ —Ç–∏–ø–∞ –∫–æ–¥–∏—Ä–∞—â–∏:

1. –û—Ä–¥–∏–Ω–∞–ª–µ–Ω –∫–æ–¥–∏—Ä–∞—â: –ø–æ–¥—Ö–æ–¥—è—â –∑–∞ –æ—Ä–¥–∏–Ω–∞–ª–Ω–∏ –ø—Ä–æ–º–µ–Ω–ª–∏–≤–∏, –∫–æ–∏—Ç–æ —Å–∞ –∫–∞—Ç–µ–≥–æ—Ä–∏—á–Ω–∏ –ø—Ä–æ–º–µ–Ω–ª–∏–≤–∏, –ø—Ä–∏ –∫–æ–∏—Ç–æ –¥–∞–Ω–Ω–∏—Ç–µ —Å–ª–µ–¥–≤–∞—Ç –ª–æ–≥–∏—á–µ—Å–∫–∞ –ø–æ–¥—Ä–µ–¥–±–∞, –∫–∞—Ç–æ –∫–æ–ª–æ–Ω–∞—Ç–∞ `–†–∞–∑–º–µ—Ä –Ω–∞ –∞—Ä—Ç–∏–∫—É–ª–∞` –≤ –Ω–∞—à–∏—è –Ω–∞–±–æ—Ä –æ—Ç –¥–∞–Ω–Ω–∏. –¢–æ–π —Å—ä–∑–¥–∞–≤–∞ –∫–∞—Ä—Ç–æ–≥—Ä–∞—Ñ–∏—Ä–∞–Ω–µ, —Ç–∞–∫–∞ —á–µ –≤—Å—è–∫–∞ –∫–∞—Ç–µ–≥–æ—Ä–∏—è –¥–∞ –±—ä–¥–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–µ–Ω–∞ —Å —á–∏—Å–ª–æ, –∫–æ–µ—Ç–æ –µ —Ä–µ–¥—ä—Ç –Ω–∞ –∫–∞—Ç–µ–≥–æ—Ä–∏—è—Ç–∞ –≤ –∫–æ–ª–æ–Ω–∞—Ç–∞.

    ```python
    from sklearn.preprocessing import OrdinalEncoder

    item_size_categories = [['sml', 'med', 'med-lge', 'lge', 'xlge', 'jbo', 'exjbo']]
    ordinal_features = ['Item Size']
    ordinal_encoder = OrdinalEncoder(categories=item_size_categories)
    ```

2. –ö–∞—Ç–µ–≥–æ—Ä–∏—á–µ–Ω –∫–æ–¥–∏—Ä–∞—â: –ø–æ–¥—Ö–æ–¥—è—â –∑–∞ –Ω–æ–º–∏–Ω–∞–ª–Ω–∏ –ø—Ä–æ–º–µ–Ω–ª–∏–≤–∏, –∫–æ–∏—Ç–æ —Å–∞ –∫–∞—Ç–µ–≥–æ—Ä–∏—á–Ω–∏ –ø—Ä–æ–º–µ–Ω–ª–∏–≤–∏, –ø—Ä–∏ –∫–æ–∏—Ç–æ –¥–∞–Ω–Ω–∏—Ç–µ –Ω–µ —Å–ª–µ–¥–≤–∞—Ç –ª–æ–≥–∏—á–µ—Å–∫–∞ –ø–æ–¥—Ä–µ–¥–±–∞, –∫–∞—Ç–æ –≤—Å–∏—á–∫–∏ —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏—Å—Ç–∏–∫–∏, —Ä–∞–∑–ª–∏—á–Ω–∏ –æ—Ç `–†–∞–∑–º–µ—Ä –Ω–∞ –∞—Ä—Ç–∏–∫—É–ª–∞` –≤ –Ω–∞—à–∏—è –Ω–∞–±–æ—Ä –æ—Ç –¥–∞–Ω–Ω–∏. –¢–æ–≤–∞ –µ –∫–æ–¥–∏—Ä–∞–Ω–µ —Å –µ–¥–∏–Ω–∏—á–µ–Ω –±–∏—Ç, –∫–æ–µ—Ç–æ –æ–∑–Ω–∞—á–∞–≤–∞, —á–µ –≤—Å—è–∫–∞ –∫–∞—Ç–µ–≥–æ—Ä–∏—è –µ –ø—Ä–µ–¥—Å—Ç–∞–≤–µ–Ω–∞ —Å –±–∏–Ω–∞—Ä–Ω–∞ –∫–æ–ª–æ–Ω–∞: –∫–æ–¥–∏—Ä–∞–Ω–∞—Ç–∞ –ø—Ä–æ–º–µ–Ω–ª–∏–≤–∞ –µ —Ä–∞–≤–Ω–∞ –Ω–∞ 1, –∞–∫–æ —Ç–∏–∫–≤–∞—Ç–∞ –ø—Ä–∏–Ω–∞–¥–ª–µ–∂–∏ –∫—ä–º —Ç–∞–∑–∏ —Ä–∞–∑–Ω–æ–≤–∏–¥–Ω–æ—Å—Ç, –∏ 0 –≤ –ø—Ä–æ—Ç–∏–≤–µ–Ω —Å–ª—É—á–∞–π.

    ```python
    from sklearn.preprocessing import OneHotEncoder

    categorical_features = ['City Name', 'Package', 'Variety', 'Origin']
    categorical_encoder = OneHotEncoder(sparse_output=False)
    ```

–°–ª–µ–¥ —Ç–æ–≤–∞ `ColumnTransformer` —Å–µ –∏–∑–ø–æ–ª–∑–≤–∞ –∑–∞ –∫–æ–º–±–∏–Ω–∏—Ä–∞–Ω–µ –Ω–∞ –º–Ω–æ–∂–µ—Å—Ç–≤–æ –∫–æ–¥–∏—Ä–∞—â–∏ –≤ –µ–¥–Ω–∞ —Å—Ç—ä–ø–∫–∞ –∏ –ø—Ä–∏–ª–∞–≥–∞–Ω–µ—Ç–æ –∏–º –∫—ä–º —Å—ä–æ—Ç–≤–µ—Ç–Ω–∏—Ç–µ –∫–æ–ª–æ–Ω–∏.

```python
    from sklearn.compose import ColumnTransformer
    
    ct = ColumnTransformer(transformers=[
        ('ord', ordinal_encoder, ordinal_features),
        ('cat', categorical_encoder, categorical_features)
        ])
    
    ct.set_output(transform='pandas')
    encoded_features = ct.fit_transform(pumpkins)
```

–û—Ç –¥—Ä—É–≥–∞ —Å—Ç—Ä–∞–Ω–∞, –∑–∞ –∫–æ–¥–∏—Ä–∞–Ω–µ –Ω–∞ –µ—Ç–∏–∫–µ—Ç–∞ –∏–∑–ø–æ–ª–∑–≤–∞–º–µ –∫–ª–∞—Å–∞ `LabelEncoder` –æ—Ç scikit-learn, –∫–æ–π—Ç–æ –µ –ø–æ–º–æ—â–µ–Ω –∫–ª–∞—Å –∑–∞ –Ω–æ—Ä–º–∞–ª–∏–∑–∏—Ä–∞–Ω–µ –Ω–∞ –µ—Ç–∏–∫–µ—Ç–∏—Ç–µ, —Ç–∞–∫–∞ —á–µ —Ç–µ –¥–∞ —Å—ä–¥—ä—Ä–∂–∞—Ç —Å–∞–º–æ —Å—Ç–æ–π–Ω–æ—Å—Ç–∏ –º–µ–∂–¥—É 0 –∏ n_classes-1 (—Ç—É–∫, 0 –∏ 1).

```python
    from sklearn.preprocessing import LabelEncoder

    label_encoder = LabelEncoder()
    encoded_label = label_encoder.fit_transform(pumpkins['Color'])
```

–°–ª–µ–¥ –∫–∞—Ç–æ —Å–º–µ –∫–æ–¥–∏—Ä–∞–ª–∏ —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏—Å—Ç–∏–∫–∏—Ç–µ –∏ –µ—Ç–∏–∫–µ—Ç–∞, –º–æ–∂–µ–º –¥–∞ –≥–∏ –æ–±–µ–¥–∏–Ω–∏–º –≤ –Ω–æ–≤ dataframe `encoded_pumpkins`.

```python
    encoded_pumpkins = encoded_features.assign(Color=encoded_label)
```

### –ê–Ω–∞–ª–∏–∑ –Ω–∞ –≤—Ä—ä–∑–∫–∏—Ç–µ –º–µ–∂–¥—É –ø—Ä–æ–º–µ–Ω–ª–∏–≤–∏—Ç–µ

–°–µ–≥–∞, –∫–æ–≥–∞—Ç–æ —Å–º–µ –æ–±—Ä–∞–±–æ—Ç–∏–ª–∏ –¥–∞–Ω–Ω–∏—Ç–µ, –º–æ–∂–µ–º –¥–∞ –∞–Ω–∞–ª–∏–∑–∏—Ä–∞–º–µ –≤—Ä—ä–∑–∫–∏—Ç–µ –º–µ–∂–¥—É —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏—Å—Ç–∏–∫–∏—Ç–µ –∏ –µ—Ç–∏–∫–µ—Ç–∞, –∑–∞ –¥–∞ –¥–æ–±–∏–µ–º –ø—Ä–µ–¥—Å—Ç–∞–≤–∞ –∫–æ–ª–∫–æ –¥–æ–±—Ä–µ –º–æ–¥–µ–ª—ä—Ç —â–µ –º–æ–∂–µ –¥–∞ –ø—Ä–µ–¥—Å–∫–∞–∂–µ –µ—Ç–∏–∫–µ—Ç–∞ –≤—ä–∑ –æ—Å–Ω–æ–≤–∞ –Ω–∞ —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏—Å—Ç–∏–∫–∏—Ç–µ. –ù–∞–π-–¥–æ–±—Ä–∏—è—Ç –Ω–∞—á–∏–Ω –∑–∞ –∏–∑–≤—ä—Ä—à–≤–∞–Ω–µ –Ω–∞ —Ç–æ–∑–∏ –≤–∏–¥ –∞–Ω–∞–ª–∏–∑ –µ —á—Ä–µ–∑ –≤–∏–∑—É–∞–ª–∏–∑–∏—Ä–∞–Ω–µ –Ω–∞ –¥–∞–Ω–Ω–∏—Ç–µ. –©–µ –∏–∑–ø–æ–ª–∑–≤–∞–º–µ –æ—Ç–Ω–æ–≤–æ —Ñ—É–Ω–∫—Ü–∏—è—Ç–∞ `catplot` –æ—Ç Seaborn, –∑–∞ –¥–∞ –≤–∏–∑—É–∞–ª–∏–∑–∏—Ä–∞–º–µ –≤—Ä—ä–∑–∫–∏—Ç–µ –º–µ–∂–¥—É `–†–∞–∑–º–µ—Ä –Ω–∞ –∞—Ä—Ç–∏–∫—É–ª–∞`, `–†–∞–∑–Ω–æ–≤–∏–¥–Ω–æ—Å—Ç` –∏ `–¶–≤—è—Ç` –≤ –∫–∞—Ç–µ–≥–æ—Ä–∏—á–µ–Ω –≥—Ä–∞—Ñ–∏–∫. –ó–∞ –ø–æ-–¥–æ–±—Ä–æ –≤–∏–∑—É–∞–ª–∏–∑–∏—Ä–∞–Ω–µ –Ω–∞ –¥–∞–Ω–Ω–∏—Ç–µ —â–µ –∏–∑–ø–æ–ª–∑–≤–∞–º–µ –∫–æ–¥–∏—Ä–∞–Ω–∞—Ç–∞ –∫–æ–ª–æ–Ω–∞ `–†–∞–∑–º–µ—Ä –Ω–∞ –∞—Ä—Ç–∏–∫—É–ª–∞` –∏ –Ω–µ–∫–æ–¥–∏—Ä–∞–Ω–∞—Ç–∞ –∫–æ–ª–æ–Ω–∞ `–†–∞–∑–Ω–æ–≤–∏–¥–Ω–æ—Å—Ç`.
